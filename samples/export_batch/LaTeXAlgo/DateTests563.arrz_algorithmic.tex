\begin{algorithm}
\caption{isLeapYear(year)}
\begin{algorithmic}[5]

\STATE \COMMENT{ Detects whether the given year is a leap year in the Gregorian calendar }
\STATE \COMMENT{ (extrapolated backwards beyonds its inauguration) }
  \STATE \(isLeapYear\gets\FALSE\)
  \COMMENT{Most years aren'{}t leap years...}
  \IF{\((year\bmod\ 4=0)\ \AND(year\bmod\ 100\neq\ 0)\)}
    \STATE \(isLeapYear\gets\TRUE\)
    \COMMENT{This is a standard leap year}
  \ELSE
    \IF{\(year\bmod\ 400=0\)}
      \STATE \COMMENT{ One of the rare leap years }
      \STATE \COMMENT{ occurring every 400 years }
      \STATE \(isLeapYear\gets\TRUE\)
    \ENDIF
  \ENDIF

\end{algorithmic}
\end{algorithm}
