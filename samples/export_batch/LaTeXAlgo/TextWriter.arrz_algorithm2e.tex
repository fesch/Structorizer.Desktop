\begin{procedure}
\caption{polygonPart(a, n, ctrclkws, nEdges, color)}
\SetKwFunction{Fnforward}{forward}
\tcc{ Draws nEdges edges of a regular n-polygon with edge length a }
\tcc{ counter-clockwise, if ctrclkws is true, or clockwise if ctrclkws is false. }
\Proc{\FuncSty{polygonPart(}\ArgSty{a, n, ctrclkws, nEdges, color}\FuncSty{)}}{
\KwData{\(a\): double}
\KwData{\(n\): integer}
\KwData{\(ctrclkws\): boolean}
\KwData{\(nEdges\): integer}
\KwData{\(color\): int}
  \(rotAngle\leftarrow{}360.0/n\)\;
  \If{\(ctrclkws\)}{
    \(rotAngle\leftarrow{}-rotAngle\)\;
  }
  \For{\(k\leftarrow 1\) \KwTo \(nEdges\) \textbf{by} \(1\)}{
    \(right(rotAngle)\)\;
    \(\Fnforward(a,color)\)\;
  }
}
\end{procedure}
