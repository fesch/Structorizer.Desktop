\begin{algorithm}
\caption{isLeapYear(1)}
\begin{algorithmic}[5]
\Function{isLeapYear}{year}
\State \Comment{ Detects whether the given year is a leap year in the Gregorian calendar }
\State \Comment{ (extrapolated backwards beyonds its inauguration) }
  \Decl{Parameters:}
    \State year: ?
  \EndDecl
  \Decl{Result type:}
    \State boolean
  \EndDecl
  \State \(isLeapYear\gets\ false\)
  \Comment{Most years aren'{}t leap years...}
  \If{\((year\bmod\ 4=0)\wedge(year\bmod\ 100\neq\ 0)\)}
    \State \(isLeapYear\gets\ true\)
    \Comment{This is a standard leap year}
  \Else
    \If{\(year\bmod\ 400=0\)}
      \State \Comment{ One of the rare leap years }
      \State \Comment{ occurring every 400 years }
      \State \(isLeapYear\gets\ true\)
    \EndIf
  \EndIf
\EndFunction
\end{algorithmic}
\end{algorithm}


\begin{algorithm}
\caption{daysInMonth423(1)}
\begin{algorithmic}[5]
\Function{daysInMonth423}{aDate}
\State \Comment{ Computes the number of days the given month (1..12) }
\State \Comment{ has in the the given year }
  \Decl{Parameters:}
    \State aDate: Date
  \EndDecl
  \Decl{Result type:}
    \State int
  \EndDecl
  \Decl{Includes:}
    \State CommonTypes423
  \EndDecl
  \Case{aDate.month}\Comment{select the case where illegal values are also considered}
    \Selector{1,3,5,7,8,10,12}
      \State \(days\gets\ 31\)
    \EndSelector
    \Selector{4,6,9,11}
      \State \(days\gets\ 30\)
    \EndSelector
    \Selector{2}
      \State \(days\gets\ 28\)
      \Comment{Default value for February}
      \State \Comment{ To make the call work it has to be done in }
      \State \Comment{ a separate element (cannot be performed }
      \State \Comment{ as part of the condition of an Alternative) }
      \State \(isLeap\gets\Call{isLeapYear}{aDate.year}\)
      \If{\(isLeap\)}
        \State \(days\gets\ 29\)
      \EndIf
    \EndSelector
    \Other
      \State \Comment{ This is the return value for illegal months. }
      \State \Comment{ It is easy to check }
      \State \(days\gets\ 0\)
    \EndOther
  \EndCase
  \State \textbf{return} \(days\)
\EndFunction
\end{algorithmic}
\end{algorithm}
