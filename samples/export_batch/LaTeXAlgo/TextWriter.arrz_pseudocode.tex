\begin{pseudocode}{digit4}{h, colorNo }
\label{digit4}
\COMMENT{ Draws digit 4 in the colour specified by colorNo with font height h }\\
\COMMENT{ from the current turtle position. }\\
\PROCEDURE{digit4}{h, colorNo}
  \COMMENT{ Octagon edge length }\\
  b\gets\ h*0.5/(sqrt(2.0)+1)\\
  \COMMENT{ Cathetus of the corner triangle outside the octagon }\\
  c\gets\ b/sqrt(2.0)\\
  \COMMENT{ inner angle at top of the triangle }\\
  angle\gets\ toDegrees(atan(1-2.0*c/h))\\
  right(90)\\
  penUp()\\
  forward(c+b)\\
  penDown()\\
  left(90)\\
  \CALL{forward}{h,colorNo}\\
  left(180-angle)\\
  \CALL{forward}{sqrt(h*h/4.0+sqr(h/2.0-c)),colorNo}\\
  left(90+angle)\\
  \CALL{forward}{h/2.0,colorNo}\\
  penUp()\\
  left(90)\\
  backward(h/2.0)\\
  penDown()\\
\ENDPROCEDURE
\end{pseudocode}

\begin{pseudocode}{polygonPart}{a, n, ctrclkws, nEdges, color }
\label{polygonPart}
\COMMENT{ Draws nEdges edges of a regular n-polygon with edge length a }\\
\COMMENT{ counter-clockwise, if ctrclkws is true, or clockwise if ctrclkws is false. }\\
\PROCEDURE{polygonPart}{a, n, ctrclkws, nEdges, color}
  rotAngle\gets\ 360.0/n\\
  \IF ctrclkws \THEN
    rotAngle\gets-rotAngle\\
  \FOR k \gets 1 \TO nEdges  \DO
  \BEGIN
    right(rotAngle)\\
    \CALL{forward}{a,color}\\
  \END\\
\ENDPROCEDURE
\end{pseudocode}

  \COMMENT{ Octagon edge length (here: edge lengzh of the square) }\\

\begin{pseudocode}{drawText}{text, h, c }
\label{drawText}
\COMMENT{ Has the turtle draw the given string '{}text\textasciiacute{} with font height '{}h\textasciiacute{} (in }\\
\COMMENT{ pixels) and the colour coded by integer '{}c\textasciiacute{} from the current Turtle }\\
\COMMENT{ position to the Turtle canvas. If the turtle looks North then }\\
\COMMENT{ the text will be written rightwards. In the event, the turtle will be }\\
\COMMENT{ placed behind the text in original orientation (such that the next text }\\
\COMMENT{ would be written like a continuation. Colour codes: }\\
\COMMENT{ 1 = black }\\
\COMMENT{ 2 = red }\\
\COMMENT{ 3 = yellow }\\
\COMMENT{ 4 = green }\\
\COMMENT{ 5 = cyan }\\
\COMMENT{ 6 = blue }\\
\COMMENT{ 7 = pink }\\
\COMMENT{ 8 = grey }\\
\COMMENT{ 9 = orange }\\
\COMMENT{ 10 = violet }\\
\COMMENT{ All letters (ASCII) will be converted to uppercase, }\\
\COMMENT{ the set of representable special characters is: decimal digits, }\\
\COMMENT{ '{}.'{}, '{},'{}, '{}!'{}, '{}?'{}, '{}"A'{}, '{}"O'{}, '{}"U'{}. Other characters will be shown as a small }\\
\COMMENT{ centred square (dummy character). }\\
\PROCEDURE{drawText}{text, h, c}
  gap\gets\ h/10.0\\
  \FOR k \gets 1 \TO length(text)  \DO
  \BEGIN
    letter\gets\ uppercase(copy(text,k,1))\\
    \IF letter=\)"{}A"{}\( \THEN
      \CALL{letterA}{h,c}\\
    \ELSEIF letter=\)"{}B"{}\( \THEN
      \CALL{letterB}{h,c}\\
    \ELSEIF letter=\)"{}C"{}\( \THEN
      \CALL{letterC}{h,c}\\
    \ELSEIF letter=\)"{}D"{}\( \THEN
      \CALL{letterD}{h,c}\\
    \ELSEIF letter=\)"{}E"{}\( \THEN
      \CALL{letterE}{h,c}\\
    \ELSEIF letter=\)"{}F"{}\( \THEN
      \CALL{letterF}{h,c}\\
    \ELSEIF letter=\)"{}G"{}\( \THEN
      \CALL{letterG}{h,c}\\
    \ELSEIF letter=\)"{}H"{}\( \THEN
      \CALL{letterH}{h,c}\\
    \ELSEIF letter=\)"{}I"{}\( \THEN
      \CALL{letterI}{h,c}\\
    \ELSEIF letter=\)"{}J"{}\( \THEN
      \CALL{letterJ}{h,c}\\
    \ELSEIF letter=\)"{}K"{}\( \THEN
      \CALL{letterK}{h,c}\\
    \ELSEIF letter=\)"{}L"{}\( \THEN
      \CALL{letterL}{h,c}\\
    \ELSEIF letter=\)"{}M"{}\( \THEN
      \CALL{letterM}{h,c}\\
    \ELSEIF letter=\)"{}N"{}\( \THEN
      \CALL{letterN}{h,c}\\
