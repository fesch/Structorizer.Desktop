\begin{struktogramm}(-14,0)[bubbleSort]
  \assign{%
    \begin{declaration}[Parameters:]
      \description{\pVar{values}}{type: \(null\)}
    \end{declaration}
  }
  \assign{\(ende\ \gets\ length(values)\ -\ 2\)}
  \until{\(posSwapped\ <\ 0\)}
    \assign{\(posSwapped\ \gets\ -1\)}
    \while{\pKey{for}\(i\ \gets\ 0\)\pKey{to}\(ende\)}
      \ifthenelse{6}{6}{\(values[i]\ >\ values[i+1]\)}{T}{F}
        \assign{\(temp\ \gets\ values[i]\)}
        \assign{\(values[i]\ \gets\ values[i+1]\)}
        \assign{\(values[i+1]\ \gets\ temp\)}
        \assign{\(posSwapped\ \gets\ i\)}
      \change
      \ifend
    \whileend
    \assign{\(ende\ \gets\ posSwapped\ -\ 1\)}
  \untilend
\end{struktogramm}

\begin{struktogramm}(-14,0)[maxHeapify]
  \assign{%
    \begin{declaration}[Parameters:]
      \description{\pVar{heap}}{type: \(null\)}
      \description{\pVar{i}}{type: \(null\)}
      \description{\pVar{range}}{type: \(null\)}
    \end{declaration}
  }
  \assign{\(right\ \gets\ (i+1)\ *\ 2\)}
  \assign{\(left\ \gets\ right\ -\ 1\)}
  \assign{\(max\ \gets\ i\)}
  \ifthenelse{6}{6}{\(left\ <\ range\ \wedge\ heap[left]\ >\ heap[i]\)}{T}{F}
    \assign{\(max\ \gets\ left\)}
  \change
  \ifend
  \ifthenelse{6}{6}{\(right\ <\ range\ \wedge\ heap[right]\ >\ heap[max]\)}{T}{F}
    \assign{\(max\ \gets\ right\)}
  \change
  \ifend
  \ifthenelse{6}{6}{\(max\ \neq\ i\)}{T}{F}
    \assign{\(temp\ \gets\ heap[i]\)}
    \assign{\(heap[i]\ \gets\ heap[max]\)}
    \assign{\(heap[max]\ \gets\ temp\)}
    \sub{\(maxHeapify(heap,\ max,\ range)\)}
  \change
  \ifend
\end{struktogramm}

\begin{struktogramm}(-14,0)[partition]
  \assign{%
    \begin{declaration}[Parameters:]
      \description{\pVar{values}}{type: \(null\)}
      \description{\pVar{start}}{type: \(null\)}
      \description{\pVar{stop}}{type: \(null\)}
      \description{\pVar{p}}{type: \(null\)}
    \end{declaration}
    \begin{declaration}[Result type:]
      \description{int}{}
    \end{declaration}
  }
  \assign{\(pivot\ \gets\ values[p]\)}
  \assign{\(values[p]\ \gets\ values[start]\)}
  \assign{\(values[start]\ \gets\ pivot\)}
  \assign{\(p\ \gets\ start\)}
  \assign{\(start\ \gets\ start\ +\ 1\)}
  \assign{\(stop\ \gets\ stop\ -\ 1\)}
  \while{\(start\ \leq\ stop\)}
    \assign{\(seen\ \gets\ values[start]\)}
    \ifthenelse{6}{6}{\(seen\ \leq\ pivot\)}{T}{F}
      \assign{\(values[p]\ \gets\ seen\)}
      \assign{\(values[start]\ \gets\ pivot\)}
      \assign{\(p\ \gets\ p\ +\ 1\)}
      \assign{\(start\ \gets\ start\ +\ 1\)}
    \change
      \assign{\(values[start]\ \gets\ values[stop]\)}
      \assign{\(values[stop]\ \gets\ seen\)}
      \assign{\(stop\ \gets\ stop\ -\ 1\)}
    \ifend
  \whileend
  \exit{\(\)\pKey{return}\(\ p\)}
\end{struktogramm}

\begin{struktogramm}(-14,0)[testSorted]
  \assign{%
    \begin{declaration}[Parameters:]
      \description{\pVar{numbers}}{type: \(null\)}
    \end{declaration}
    \begin{declaration}[Result type:]
      \description{bool}{}
    \end{declaration}
  }
  \assign{\(isSorted\ \gets\ true\)}
  \assign{\(i\ \gets\ 0\)}
  \while{\(isSorted\ \wedge\ (i\ \leq\ length(numbers)-2)\)}
    \ifthenelse{6}{6}{\(numbers[i]\ >\ numbers[i+1]\)}{T}{F}
      \assign{\(isSorted\ \gets\ false\)}
    \change
      \assign{\(i\ \gets\ i\ +\ 1\)}
    \ifend
  \whileend
  \exit{\(\)\pKey{return}\(\ isSorted\)}
\end{struktogramm}

\begin{struktogramm}(-14,0)[buildMaxHeap]
  \assign{%
    \begin{declaration}[Parameters:]

    \end{declaration}
  }
  \assign{\(lgth\ \gets\ length(heap)\)}
  \while{\pKey{for}\(k\ \gets\ lgth\ div\ 2\ -\ 1\)\pKey{to}\(0\)\pKey{by}\(-1\)}
    \sub{\(maxHeapify(heap,\ k,\ lgth)\)}
  \whileend
\end{struktogramm}

\begin{struktogramm}(-14,0)[quickSort]
  \assign{%
    \begin{declaration}[Parameters:]
      \description{\pVar{values}}{type: \(null\)}
      \description{\pVar{start}}{type: \(null\)}
      \description{\pVar{stop}}{type: \(null\)}
    \end{declaration}
  }
  \ifthenelse{6}{6}{\(stop\ \geq\ start\ +\ 2\)}{T}{F}
    \assign{\(p\ \gets\ random(stop-start)\ +\ start\)}
    \sub{\(p\ \gets\ partition(values,\ start,\ stop,\ p)\)}
    \inparallel{2} {quickSort(values,\ start,\ p)}
    \task{quickSort(values,\ p+1,\ stop)}
    \inparallelend
  \change
  \ifend
\end{struktogramm}

\begin{struktogramm}(-14,0)[heapSort]
  \assign{%
    \begin{declaration}[Parameters:]
      \description{\pVar{values}}{type: \(null\)}
    \end{declaration}
  }
  \sub{\(buildMaxHeap(values)\)}
  \assign{\(heapRange\ \gets\ length(values)\)}
  \while{\pKey{for}\(k\ \gets\ heapRange\ -\ 1\)\pKey{to}\(1\)\pKey{by}\(-1\)}
    \assign{\(heapRange\ \gets\ heapRange\ -\ 1\)}
    \assign{\(maximum\ \gets\ values[0]\)}
    \assign{\(values[0]\ \gets\ values[heapRange]\)}
    \assign{\(values[heapRange]\ \gets\ maximum\)}
    \sub{\(maxHeapify(values,\ 0,\ heapRange)\)}
  \whileend
\end{struktogramm}

\begin{struktogramm}(-14,0)[SORTING\_TEST\_MAIN]
  \until{\(elementCount\ \geq\ 1\)}
    \assign{\(\)\pKey{INPUT}\(\ elementCount\)}
  \untilend
  \until{\(modus\ =\ 1\ \vee\ modus\ =\ 2\ \vee\ modus\ =\ 3\)}
    \assign{\(\)\pKey{INPUT}\(\ \)"{}Filling:\ 1\ =\ random,\ 2\ =\ increasing,\ 3\ =\ decreasing"{}\(,\ modus\)}
  \untilend
  \while{\pKey{for}\(i\ \gets\ 0\)\pKey{to}\(elementCount-1\)}
    \case{6}{4}{\(modus\)}{1}
        \assign{\(values1[i]\ \gets\ random(10000)\)}
      \switch{2}
        \assign{\(values1[i]\ \gets\ i\)}
      \switch{3}
        \assign{\(values1[i]\ \gets\ -i\)}
     \caseend
  \whileend
  \while{\pKey{for}\(i\ \gets\ 0\)\pKey{to}\(elementCount-1\)}
    \assign{\(values2[i]\ \gets\ values1[i]\)}
    \assign{\(values3[i]\ \gets\ values1[i]\)}
  \whileend
  \inparallel{3} {bubbleSort(values1)}
  \task{quickSort(values2,\ 0,\ elementCount)}
  \task{heapSort(values3)}
  \inparallelend
  \sub{\(ok1\ \gets\ testSorted(values1)\)}
  \sub{\(ok2\ \gets\ testSorted(values2)\)}
  \sub{\(ok3\ \gets\ testSorted(values3)\)}
  \ifthenelse{6}{6}{\(!\ ok1\ \vee\ !\ ok2\ \vee\ !\ ok3\)}{T}{F}
    \while{\pKey{for}\(i\ \gets\ 0\)\pKey{to}\(elementCount-1\)}
      \ifthenelse{6}{6}{\(values1[i]\ \neq\ values2[i]\ \vee\ values1[i]\ \neq\ values3[i]\)}{T}{F}
        \assign{\(\)\pKey{OUTPUT}\(\ \)"{}Difference\ at\ ["{}\(,\ i,\ \)"{}]:\ "{}\(,\ values1[i],\ \)"{}\ \textless-\textgreater\ "{}\(,\ values2[i],\ \)"{}\ \textless-\textgreater\ "{}\(,\ values3[i]\)}
      \change
      \ifend
    \whileend
  \change
  \ifend
  \until{\(show\ =\ \)"{}yes"{}\(\ \vee\ show\ =\ \)"{}no"{}\(\)}
    \assign{\(\)\pKey{INPUT}\(\ \)"{}Show\ arrays\ (yes/no)?"{}\(,\ show\)}
  \untilend
  \ifthenelse{6}{6}{\(show\ =\ \)"{}yes"{}\(\)}{T}{F}
    \while{\pKey{for}\(i\ \gets\ 0\)\pKey{to}\(elementCount\ -\ 1\)}
      \assign{\(\)\pKey{OUTPUT}\(\ \)"{}["{}\(,\ i,\ \)"{}]:\textbackslash{}t"{}\(,\ values1[i],\ \)"{}\textbackslash{}t"{}\(,\ values2[i],\ \)"{}\textbackslash{}t"{}\(,\ values3[i]\)}
    \whileend
  \change
  \ifend
\end{struktogramm}
